Untold Lang is a powerful, expressive programming language built for AI, Web, Apps, Scripting, Security and more — all with one clean syntax.

Try in Browser Read the Docs 
use untold.ai
use untold.web

start main() {
    // AI-powered web scraper in 8 lines
    let res      = http.get("https://example.com")
    let keywords = ai.keywords(res.body, 5)
    let mood     = ai.sentiment(res.body)

    say("Keywords: " + text(keywords))
    say("Sentiment: " + mood.label)
}
